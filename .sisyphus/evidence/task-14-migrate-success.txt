```typescript
export function buildLegacyScreenplaySegment(
  scene: Pick<ScriptSegmentDto, 'sceneNo' | 'action' | 'dialogue' | 'emotion'>
): Pick<ScriptSegmentDto, 'screenplay' | 'screenplayScenes' | 'action' | 'dialogue' | 'emotion' | 'legacyFormat'> {
  const screenplay = buildScreenplayFromStructuredScene({
    ...scene,
    screenplay: '',
    screenplayScenes: []
  })

  return {
    screenplay,
    screenplayScenes: parseScreenplayScenes(screenplay),
    legacyFormat: true,  // Mark as legacy-migrated
    action: scene.action,
    dialogue: scene.dialogue,
    emotion: scene.emotion
  }
}
```
